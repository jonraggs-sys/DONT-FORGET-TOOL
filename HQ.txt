import React, { useState, useEffect } from 'react';
import { initializeApp } from 'firebase/app';
import { 
  getAuth, 
  signInAnonymously, 
  signInWithCustomToken, 
  onAuthStateChanged 
} from 'firebase/auth';
import { 
  getFirestore, 
  collection, 
  addDoc, 
  doc, 
  updateDoc, 
  deleteDoc, 
  onSnapshot 
} from 'firebase/firestore';

// ============================================================================
// FIREBASE CONFIGURATION
// Currently configured to run dynamically in this preview environment.
// WHEN DEPLOYING TO NETLIFY/GITHUB: Replace this block with your actual keys
// from the Firebase Console (Project Settings -> General -> Your apps).
// ============================================================================
const firebaseConfig = typeof __firebase_config !== 'undefined' 
  ? JSON.parse(__firebase_config) 
  : {
      // Paste your real keys here before pushing to GitHub:
      // apiKey: "YOUR_API_KEY",
      // authDomain: "your-app.firebaseapp.com",
      // projectId: "your-project-id",
      // storageBucket: "your-project.appspot.com",
      // messagingSenderId: "123456789",
      // appId: "YOUR_APP_ID"
    };

const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);
const appId = typeof __app_id !== 'undefined' ? __app_id : 'hq-dashboard-local';

export default function App() {
  const [user, setUser] = useState(null);
  const [tasks, setTasks] = useState([]);
  const [newTask, setNewTask] = useState('');
  const [category, setCategory] = useState('Household');
  const [priority, setPriority] = useState('Medium');
  const [deadline, setDeadline] = useState('');
  const [loading, setLoading] = useState(true);

  // 1. Initialize Authentication
  useEffect(() => {
    const initAuth = async () => {
      try {
        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
          await signInWithCustomToken(auth, __initial_auth_token);
        } else {
          await signInAnonymously(auth);
        }
      } catch (error) {
        console.error("Authentication Error:", error);
      }
    };
    
    initAuth();
    
    const unsubscribe = onAuthStateChanged(auth, (currentUser) => {
      setUser(currentUser);
      setLoading(false);
    });
    
    return () => unsubscribe();
  }, []);

  // 2. Fetch Tasks from Database
  useEffect(() => {
    if (!user) return; // Guard: Wait for auth

    // Using the required structure for this environment's private data
    const tasksRef = collection(db, 'artifacts', appId, 'users', user.uid, 'tasks');
    
    const unsubscribe = onSnapshot(tasksRef, (snapshot) => {
      const fetchedTasks = snapshot.docs.map(doc => ({
        id: doc.id,
        ...doc.data()
      }));
      
      // Sort in memory (newest first)
      fetchedTasks.sort((a, b) => b.createdAt - a.createdAt);
      setTasks(fetchedTasks);
    }, (error) => {
      console.error("Database fetch error:", error);
    });

    return () => unsubscribe();
  }, [user]);

  // 3. Add Task
  const addTask = async (e) => {
    e.preventDefault();
    if (!newTask.trim() || !user) return;
    
    const tasksRef = collection(db, 'artifacts', appId, 'users', user.uid, 'tasks');
    
    await addDoc(tasksRef, {
      text: newTask,
      category,
      priority,
      deadline,
      completed: false,
      createdAt: Date.now()
    });
    
    setNewTask('');
    setDeadline('');
    // Note: State updates automatically via onSnapshot listener
  };

  // 4. Toggle Complete Status
  const toggleComplete = async (taskId, currentStatus) => {
    if (!user) return;
    const taskDoc = doc(db, 'artifacts', appId, 'users', user.uid, 'tasks', taskId);
    await updateDoc(taskDoc, { completed: !currentStatus });
  };

  // 5. Delete Task
  const deleteTask = async (taskId) => {
    if (!user) return;
    const taskDoc = doc(db, 'artifacts', appId, 'users', user.uid, 'tasks', taskId);
    await deleteDoc(taskDoc);
  };

  const getPriorityColor = (level) => {
    if (level === 'High') return 'text-red-500 bg-red-100 border border-red-200';
    if (level === 'Medium') return 'text-yellow-700 bg-yellow-100 border border-yellow-200';
    return 'text-green-700 bg-green-100 border border-green-200';
  };

  if (loading) {
    return (
      <div className="min-h-screen bg-gray-50 flex items-center justify-center">
        <div className="animate-pulse text-gray-500 font-medium">Booting HQ Dashboard...</div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-50 p-4 md:p-8 font-sans">
      <div className="max-w-3xl mx-auto bg-white rounded-2xl shadow-sm border border-gray-200 overflow-hidden">
        
        {/* Header */}
        <div className="bg-slate-900 p-6 sm:p-8 text-white">
          <div className="flex justify-between items-center">
            <div>
              <h1 className="text-2xl sm:text-3xl font-bold tracking-tight text-white">HQ Dashboard</h1>
              <p className="text-slate-400 text-sm mt-1">Daily Operations & Task Management</p>
            </div>
            <div className="text-xs text-slate-500 bg-slate-800 px-3 py-1 rounded-full border border-slate-700">
              Live Sync Active
            </div>
          </div>
        </div>

        <div className="p-6 sm:p-8">
          {/* Input Form */}
          <form onSubmit={addTask} className="space-y-4 mb-8 bg-slate-50 p-5 rounded-xl border border-slate-200">
            <div>
              <input
                type="text"
                value={newTask}
                onChange={(e) => setNewTask(e.target.value)}
                placeholder="What needs to get done?"
                className="w-full p-3.5 border border-slate-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-slate-900 focus:border-transparent transition-all shadow-sm"
              />
            </div>
            
            <div className="flex flex-col sm:flex-row gap-3">
              <select 
                value={category} 
                onChange={(e) => setCategory(e.target.value)}
                className="p-3 border border-slate-300 rounded-lg bg-white text-sm shadow-sm focus:ring-2 focus:ring-slate-900 focus:outline-none"
              >
                <option>Household</option>
                <option>Errand</option>
                <option>Personal</option>
              </select>

              <select 
                value={priority} 
                onChange={(e) => setPriority(e.target.value)}
                className="p-3 border border-slate-300 rounded-lg bg-white text-sm shadow-sm focus:ring-2 focus:ring-slate-900 focus:outline-none"
              >
                <option>Low Priority</option>
                <option>Medium</option>
                <option>High Priority</option>
              </select>

              <input 
                type="date" 
                value={deadline}
                onChange={(e) => setDeadline(e.target.value)}
                className="p-3 border border-slate-300 rounded-lg bg-white text-sm shadow-sm focus:ring-2 focus:ring-slate-900 focus:outline-none text-slate-700"
              />

              <button 
                type="submit" 
                className="sm:ml-auto bg-slate-900 text-white px-6 py-3 rounded-lg hover:bg-slate-800 transition-colors shadow-md text-sm font-semibold tracking-wide"
              >
                Add Task
              </button>
            </div>
          </form>

          {/* Task List */}
          <div className="space-y-3">
            {tasks.length === 0 ? (
              <div className="text-center py-12 border-2 border-dashed border-slate-200 rounded-xl">
                <p className="text-slate-500 font-medium">Dashboard is clear.</p>
                <p className="text-slate-400 text-sm mt-1">Add a task above to begin.</p>
              </div>
            ) : (
              tasks.map(task => (
                <div 
                  key={task.id} 
                  className={`group flex flex-col sm:flex-row sm:items-center justify-between p-4 border rounded-xl transition-all duration-200 ${
                    task.completed 
                      ? 'bg-slate-50 border-slate-200 opacity-60' 
                      : 'bg-white border-slate-200 hover:border-slate-300 shadow-sm hover:shadow'
                  }`}
                >
                  <div className="flex items-start sm:items-center gap-4">
                    <input 
                      type="checkbox" 
                      checked={task.completed}
                      onChange={() => toggleComplete(task.id, task.completed)}
                      className="mt-1 sm:mt-0 w-5 h-5 cursor-pointer accent-slate-900 rounded border-slate-300"
                    />
                    <div>
                      <p className={`font-semibold text-base ${task.completed ? 'line-through text-slate-500' : 'text-slate-900'}`}>
                        {task.text}
                      </p>
                      <div className="flex flex-wrap gap-2 mt-2 text-xs">
                        <span className="text-slate-600 font-medium px-2.5 py-1 bg-slate-100 rounded-md border border-slate-200">
                          {task.category}
                        </span>
                        <span className={`font-medium px-2.5 py-1 rounded-md ${getPriorityColor(task.priority)}`}>
                          {task.priority}
                        </span>
                        {task.deadline && (
                          <span className="text-slate-600 font-medium px-2.5 py-1 bg-slate-100 rounded-md border border-slate-200 flex items-center gap-1">
                            <svg className="w-3 h-3" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M8 7V3m8 4V3m-9 8h10M5 21h14a2 2 0 002-2V7a2 2 0 00-2-2H5a2 2 0 00-2 2v12a2 2 0 002 2z"></path></svg>
                            {task.deadline}
                          </span>
                        )}
                      </div>
                    </div>
                  </div>
                  
                  <button 
                    onClick={() => deleteTask(task.id)}
                    className="mt-4 sm:mt-0 self-end sm:self-auto text-slate-400 hover:text-red-500 hover:bg-red-50 p-2 rounded-lg transition-colors font-medium text-sm"
                  >
                    Delete
                  </button>
                </div>
              ))
            )}
          </div>
        </div>
      </div>
    </div>
  );
}
